# 상수를 사용하는 이유

- 문자열을 다른 문자열로 고치고 싶을 때, 사용된 모든 위치를 찾아서 바꿔 줄 필요 없이 상수를 선언한 제일 상단부분만 수정하면 되는 이점
- 문자열이 아닌, 코드 형태의 변수명으로 작성할 경우 오타가 나면 브라우저가 바로 에러 메시지로 알려 줄 수 있다는 점 

```javascript

var SCISSOR = '가위'; 

if (userInput === '가') {
    // 원래 '가위'가 맞아도 오류 메시지가 안뜸 
}

if (userInput === SISSOR) {
    // 상수가 오타가 나면 브라우저가 오류를 알려줌 
}
